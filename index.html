SELECT

Reviews,

CASE

WHEN Reviews LIKE '%worst%'

or Reviews LIKE '%disappoint%'

or Reviews LIKE '%bad%'

or Reviews LIKE '%Terrible%'

or Reviews LIKE '%poor%' THEN 'B' -- bad satisfaction

WHEN Reviews LIKE '%good%'

or Reviews LIKE '%nice%' THEN 'G' -- good satisfaction

ELSE 'N' -- neutral

END AS satisfaction_tag,

CASE

WHEN Reviews LIKE '%attentive%'

or Reviews LIKE '%helpful%'

or Reviews LIKE '%excellent%'

or Reviews LIKE '%comfortable%'

or Reviews LIKE '%pleasant%' THEN 'H' -- good service

WHEN Reviews LIKE '%late%' or Reviews LIKE '%cancel%' or Reviews LIKE '%uncomfortable%' or Reviews LIKE '%chaotic%' THEN 'L' -- Low service quality

ELSE 'M' -- neutral

END AS service_tag,

CASE

WHEN Reviews LIKE '%nice%' or Reviews LIKE '%amazing%' or Reviews LIKE '%good%' or Reviews LIKE '%happy%' THEN 'H' -- positive response

WHEN Reviews LIKE '%disappoin%' or Reviews LIKE '%sad%' THEN 'S' -- negative response

ELSE 'R' -- general

END AS emotional_tag,

CASE

WHEN Reviews LIKE '%turbule%' or Reviews LIKE '%injur%' or Reviews Like "%delayed flight%" THEN 'A' -- critical issue exist

ELSE 'X' -- no critical issue

END AS critical_Issues_tag

FROM airlines_reviews

WHERE Airline = "Singapore Airlines";
